peliculaspormesyanno(mes,anno) { var objetoaenviar = this; var vectordepeliculasfiltradas = []; // Return a new promise. return new Promise(function(resolve, reject) { // Do the usual XHR stuff try { var xhr = new XMLHttpRequest(); xhr.open('POST', 'http://localhost:8080/api/seleccionartodos'); xhr.setRequestHeader('Content-Type', 'application/json'); xhr.onload = function() { if (xhr.status === 200) { var peliculas =JSON.parse(xhr.responseText); for(var elemento in peliculas) { var fechadelapelicula = new Date(peliculas[elemento].FECHA); if((fechadelapelicula.getMonth() == mes) && (fechadelapelicula.getFullYear() == anno)) { vectordepeliculasfiltradas.push(peliculas[elemento]); } } resolve(vectordepeliculasfiltradas); } else { reject(xhr); } }; xhr.send(JSON.stringify(objetoaenviar)); } catch(err) { reject(err.message); } }); }
